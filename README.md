# test-node-app
a simple test node js application

#to run the app add .env file to the root folder with databases properties

DB_HOST=localhost\
DB_USER=root\
DB_PASSWORD=root_pwd\
DB_NAME=testdb

run `npm install`\
run `npm run build`\
run `npm start`  - to run locally\
run `npm test` - to run integration tests
